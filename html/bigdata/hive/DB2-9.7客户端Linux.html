1. DB2-9.7空间数据库安装¶

Note

  • 数据库名称: cn_suby_material_mall

  • 数据库表前缀: suby_

CREATE DATABASE  `cn_suby_material_mall` DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

grant create,select,insert,update,delete,drop,alter on cn_suby_material_mall.* to suby_material@"%" identified by "yourpasswd";
### mysql 8
create user 'suby_material'@'%' identified with mysql_native_password by 'yourpassword';
grant all privileges on cn_suby_material_mall.* to 'suby_material'@'%';
flush privileges;
